function solution(k, tangerine) {
  let answer = 0;
  const map = {};

  tangerine.forEach((size) => {
    map[size] = (map[size] || 0) + 1;
  });

  const counts = Object.values(map).sort((a, b) => b - a);

  let sum = 0;
  for (const cnt of counts) {
    sum += cnt;
    answer += 1;
    if (sum >= k) break;
  }

  return answer;
}
